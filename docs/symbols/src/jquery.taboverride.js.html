 12 /**
 13  * the TABOVERRIDE "namespace" global object
 14  *
 15  * @namespace
 16  */
 17 var TABOVERRIDE;
 18 if (typeof TABOVERRIDE !== 'object') {
 19     TABOVERRIDE = {};
 20 }
 21 
 22 (function (TABOVERRIDE) {
 23     'use strict';
 24 
 25     var aTab = '\t', // the string representing a tab
 26         autoIndent = false, // whether each line should be automatically indented
 27         inWhitespace = false, // whether the start of the selection is in the leading whitespace on enter
 28         ta = document.createElement('textarea'), // temp textarea element to get newline character(s)
 29         newline, // the newline character sequence (\n or \r\n)
 30         newlineLen; // the number of characters used for a newline (1 or 2)
 31 
 32     /**
 33      * Inserts or removes tabs and newlines on the keyDown event for the tab or enter key.
 34      *
 35      * @param {Event} e  the event object
 36      */
 37     TABOVERRIDE.overrideKeyDown = function (e) {
 38         e = e || event;
 39 
 40         var key = e.keyCode, // the key code for the key that was pressed
 41             tab, // the string representing a tab
 42             tabLen, // the length of a tab
 43             text, // initial text in the textarea
 44             range, // the IE TextRange object
 45             tempRange, // used to calculate selection start and end positions in IE
 46             preNewlines, // the number of newline character sequences before the selection start (for IE)
 47             selNewlines, // the number of newline character sequences within the selection (for IE)
 48             initScrollTop, // initial scrollTop value used to fix scrolling in Firefox
 49             selStart, // the selection start position
 50             selEnd, // the selection end position
 51             sel, // the selected text
 52             startLine, // for multi-line selections, the first character position of the first line
 53             endLine, // for multi-line selections, the last character position of the last line
 54             numTabs, // the number of tabs inserted / removed in the selection
 55             startTab, // if a tab was removed from the start of the first line
 56             preTab, // if a tab was removed before the start of the selection
 57             whitespace, // the whitespace at the beginning of the first selected line
 58             whitespaceLen, // the length of the whitespace at the beginning of the first selected line
 59             CHARACTER = 'character'; // string constant used for the Range.move methods
 60 
 61         // don't do any unnecessary work
 62         if ((key !== 9 && (key !== 13 || !autoIndent)) || e.ctrlKey || e.altKey) {
 63             return;
 64         }
 65 
 66         // initialize variables used for tab and enter keys
 67         inWhitespace = false; // this will be set to true if enter is pressed in the leading whitespace
 68         text = this.value;
 69 
 70         // this is really just for Firefox, but will be used by all browsers that support
 71         // selectionStart and selectionEnd - whenever the textarea value property is reset,
 72         // Firefox scrolls back to the top - this is used to set it back to the original value
 73         // scrollTop is nonstandard, but supported by all modern browsers
 74         initScrollTop = this.scrollTop;
 75 
 76         // get the text selection
 77         // prefer the nonstandard document.selection way since it allows for
 78         // automatic scrolling to the cursor via the range.select() method
 79         if (document.selection) { // IE
 80             range = document.selection.createRange();
 81             sel = range.text;
 82             tempRange = range.duplicate();
 83             tempRange.moveToElementText(this);
 84             tempRange.setEndPoint('EndToEnd', range);
 85             selEnd = tempRange.text.length;
 86             selStart = selEnd - sel.length;
 87 
 88             // whenever the value of the textarea is changed, the range needs to be reset
 89             // IE <9 (and Opera) use both \r and \n for newlines - this adds an extra character
 90             // that needs to be accounted for when doing position calculations with ranges
 91             // these values are used to offset the selection start and end positions
 92             if (newlineLen > 1) {
 93                 preNewlines = text.slice(0, selStart).split(newline).length - 1;
 94                 selNewlines = sel.split(newline).length - 1;
 95             } else {
 96                 preNewlines = selNewlines = 0;
 97             }
 98         } else if (typeof this.selectionStart === 'number') {
 99             selStart = this.selectionStart;
100             selEnd = this.selectionEnd;
101             sel = text.slice(selStart, selEnd);
102         } else {
103             return; // cannot access textarea selection - do nothing
104         }
105 
106         // tab key - insert / remove tab
107         if (key === 9) {
108 
109             // initialize tab variables
110             tab = aTab;
111             tabLen = tab.length;
112             numTabs = 0;
113             startTab = 0;
114             preTab = 0;
115 
116             // multi-line selection
117             if (selStart !== selEnd && sel.indexOf('\n') !== -1) {
118                 // for multiple lines, only insert / remove tabs from the beginning of each line
119 
120                 // find the start of the first selected line
121                 if (selStart === 0 || text.charAt(selStart - 1) === '\n') {
122                     // the selection starts at the beginning of a line
123                     startLine = selStart;
124                 } else {
125                     // the selection starts after the beginning of a line
126                     // set startLine to the beginning of the first partially selected line
127                     // subtract 1 from selStart in case the cursor is at the newline character,
128                     // for instance, if the very end of the previous line was selected
129                     // add 1 to get the next character after the newline
130                     // if there is none before the selection, lastIndexOf returns -1
131                     // when 1 is added to that it becomes 0 and the first character is used
132                     startLine = text.lastIndexOf('\n', selStart - 1) + 1;
133                 }
134 
135                 // find the end of the last selected line
136                 if (selEnd === text.length || text.charAt(selEnd) === '\n') {
137                     // the selection ends at the end of a line
138                     endLine = selEnd;
139                 } else {
140                     // the selection ends before the end of a line
141                     // set endLine to the end of the last partially selected line
142                     endLine = text.indexOf('\n', selEnd);
143                     if (endLine === -1) {
144                         endLine = text.length;
145                     }
146                 }
147 
148                 // if the shift key was pressed, remove tabs instead of inserting them
149                 if (e.shiftKey) {
150                     if (text.slice(startLine).indexOf(tab) === 0) {
151                         // is this tab part of the selection?
152                         if (startLine === selStart) {
153                             // it is, remove it
154                             sel = sel.slice(tabLen);
155                         } else {
156                             // the tab comes before the selection
157                             preTab = tabLen;
158                         }
159                         startTab = tabLen;
160                     }
161 
162                     this.value = text.slice(0, startLine) + text.slice(startLine + preTab, selStart) +
163                         sel.replace(new RegExp('\n' + tab, 'g'), function () {
164                             numTabs += 1;
165                             return '\n';
166                         }) + text.slice(selEnd);
167 
168                     // set start and end points
169                     if (range) { // IE
170                         // setting end first makes calculations easier
171                         range.collapse();
172                         range.moveEnd(CHARACTER, selEnd - startTab - (numTabs * tabLen) - selNewlines - preNewlines);
173                         range.moveStart(CHARACTER, selStart - preTab - preNewlines);
174                         range.select();
175                     } else {
176                         // set start first for Opera
177                         this.selectionStart = selStart - preTab; // preTab is 0 or tabLen
178                         // move the selection end over by the total number of tabs removed
179                         this.selectionEnd = selEnd - startTab - (numTabs * tabLen);
180                     }
181                 } else { // no shift key
182                     numTabs = 1; // for the first tab
183                     // insert tabs at the beginning of each line of the selection
184                     this.value = text.slice(0, startLine) + tab + text.slice(startLine, selStart) +
185                         sel.replace(/\n/g, function () {
186                             numTabs += 1;
187                             return '\n' + tab;
188                         }) + text.slice(selEnd);
189 
190                     // set start and end points
191                     if (range) { // IE
192                         range.collapse();
193                         range.moveEnd(CHARACTER, selEnd + (numTabs * tabLen) - selNewlines - preNewlines);
194                         range.moveStart(CHARACTER, selStart + tabLen - preNewlines);
195                         range.select();
196                     } else {
197                         // the selection start is always moved by 1 character
198                         this.selectionStart = selStart + tabLen;
199                         // move the selection end over by the total number of tabs inserted
200                         this.selectionEnd = selEnd + (numTabs * tabLen);
201                         this.scrollTop = initScrollTop;
202                     }
203                 }
204             } else { // single line selection
205                 // if the shift key was pressed, remove a tab instead of inserting one
206                 if (e.shiftKey) {
207                     // if the character before the selection is a tab, remove it
208                     if (text.slice(selStart - tabLen).indexOf(tab) === 0) {
209                         this.value = text.slice(0, selStart - tabLen) + text.slice(selStart);
210 
211                         // set start and end points
212                         if (range) { // IE
213                             // collapses range and moves it by -1 tab
214                             range.move(CHARACTER, selStart - tabLen - preNewlines);
215                             range.select();
216                         } else {
217                             this.selectionEnd = this.selectionStart = selStart - tabLen;
218                             this.scrollTop = initScrollTop;
219                         }
220                     }
221                 } else { // no shift key - insert a tab
222                     if (range) { // IE
223                         range.text = tab;
224                         range.select();
225                     } else {
226                         this.value = text.slice(0, selStart) + tab + text.slice(selEnd);
227                         this.selectionEnd = this.selectionStart = selStart + tabLen;
228                         this.scrollTop = initScrollTop;
229                     }
230                 }
231             }
232         } else if (autoIndent) { // Enter key
233             // insert a newline and copy the whitespace from the beginning of the line
234 
235             // find the start of the first selected line
236             if (selStart === 0 || text.charAt(selStart - 1) === '\n') {
237                 // the selection starts at the beginning of a line
238                 // do nothing special
239                 inWhitespace = true;
240                 return;
241             }
242 
243             // see explanation under "multi-line selection" above
244             startLine = text.lastIndexOf('\n', selStart - 1) + 1;
245 
246             // find the end of the first selected line
247             endLine = text.indexOf('\n', selStart);
248 
249             // if no newline is found, set endLine to the end of the text
250             if (endLine === -1) {
251                 endLine = text.length;
252             }
253 
254             // get the whitespace at the beginning of the first selected line (spaces and tabs only)
255             whitespace = text.slice(startLine, endLine).match(/^[ \t]*/)[0];
256             whitespaceLen = whitespace.length;
257 
258             // the cursor (selStart) is in the whitespace at beginning of the line
259             // do nothing special
260             if (selStart < startLine + whitespaceLen) {
261                 inWhitespace = true;
262                 return;
263             }
264 
265             if (range) { // IE
266                 // insert the newline and whitespace
267                 range.text = '\n' + whitespace;
268                 range.select();
269             } else {
270                 // insert the newline and whitespace
271                 this.value = text.slice(0, selStart) + '\n' + whitespace + text.slice(selEnd);
272                 // Opera uses \r\n for a newline, instead of \n,
273                 // so use newlineLen instead of a hard-coded value
274                 this.selectionEnd = this.selectionStart = selStart + newlineLen + whitespaceLen;
275                 this.scrollTop = initScrollTop;
276             }
277         }
278 
279         if (e.preventDefault) {
280             e.preventDefault();
281         } else {
282             e.returnValue = false;
283             return false;
284         }
285     };
286 
287     /**
288      * Prevents the default action for the keyPress event when tab or enter is
289      * pressed. Opera (and Firefox) also fire a keypress event when the tab or
290      * enter key is pressed. Opera requires that the default action be prevented
291      * on this event or the textarea will lose focus.
292      *
293      * @param {Event} e  the event object
294      *
295      * @memberOf TABOVERRIDE
296      */
297     TABOVERRIDE.overrideKeyPress = function (e) {
298         e = e || event;
299 
300         var key = e.keyCode;
301 
302         if ((key === 9 || (key === 13 && autoIndent && !inWhitespace)) && !e.ctrlKey && !e.altKey) {
303             if (e.preventDefault) {
304                 e.preventDefault();
305             } else {
306                 e.returnValue = false;
307                 return false;
308             }
309         }
310     };
311 
312     /**
313      * Gets or sets the tab size for all elements that have Tab Override enabled.
314      * 0 represents the tab character.
315      *
316      * @param  {Number}        [size]  the tab size
317      * @return {Number|Object}         the tab size or the TABOVERRIDE object
318      *
319      * @name tabSize
320      * @function
321      * @memberOf TABOVERRIDE
322      */
323     TABOVERRIDE.tabSize = function (size) {
324         var i;
325 
326         if (arguments.length) {
327             if (!size) { // size is 0 (or falsy)
328                 aTab = '\t';
329             } else if (typeof size === 'number' && size > 0) {
330                 aTab = '';
331                 for (i = 0; i < size; i += 1) {
332                     aTab += ' ';
333                 }
334             }
335             return this;
336         }
337 
338         return (aTab === '\t') ? 0 : aTab.length;
339     };
340 
341     /**
342      * Gets or sets the auto indent setting. True if each line should be
343      * automatically indented (default = false).
344      *
345      * @param  {Boolean}        [enable]  whether auto indent should be enabled
346      * @return {Boolean|Object}           whether auto indent is enabled or the
347      *                                    TABOVERRIDE object
348      *
349      * @name autoIndent
350      * @function
351      * @memberOf TABOVERRIDE
352      */
353     TABOVERRIDE.autoIndent = function (enable) {
354         if (arguments.length) {
355             autoIndent = enable ? true : false;
356             return this;
357         }
358 
359         return autoIndent;
360     };
361 
362     // get the characters used for a newline
363     ta.value = '\n';
364     newline = ta.value;
365     newlineLen = newline.length;
366     ta = null;
367 }(TABOVERRIDE));

388 (function ($, TABOVERRIDE) {
389     'use strict';
390 
391     /**
392      * the tabOverride method - Tabs will be overridden if enable is true.
393      *
394      * @param  {Boolean} [enable=true]  whether Tab Override should be enabled
395      *                                  for the element(s)
396      * @return {Object}                 the jQuery object
397      *
398      * @name tabOverride
399      * @memberOf jQuery.fn
400      * @function
401      * @namespace
402      */
403     var tabOverride = $.fn.tabOverride = function (enable) {
404         // unbind the tab override functions so they are not bound more than once
405         this.each(function () {
406             $(this).off('.tabOverride');
407         });
408 
409         // only bind the tab override functions if the enable argument is not
410         // specified or is truthy
411         if (!arguments.length || enable) {
412             this.each(function () {
413                 if (this.nodeName && this.nodeName.toLowerCase() === 'textarea') {
414                     $(this)
415                         .on('keydown.tabOverride', TABOVERRIDE.overrideKeyDown)
416                         .on('keypress.tabOverride', TABOVERRIDE.overrideKeyPress);
417                 }
418             });
419         }
420 
421         return this; // always return the jQuery object
422     };
423 
424     /**
425      * Gets or sets the tab size for all elements that have Tab Override enabled.
426      * 0 represents the tab character.
427      *
428      * @param  {Number}          [size]  the tab size
429      * @return {Number|Function}         the tab size or the tabOverride function
430      *
431      * @name tabSize
432      * @function
433      * @memberOf jQuery.fn.tabOverride
434      */
435     tabOverride.tabSize = TABOVERRIDE.tabSize;
436 
437     /**
438      * Returns the current tab size. 0 represents the tab character.
439      *
440      * @return {Number}  the size (length) of the tab string or 0 for the tab
441      *                   character
442      *
443      * @name getTabSize
444      * @function
445      * @memberOf jQuery.fn.tabOverride
446      * @deprecated since 2.0 - use tabSize() instead
447      */
448     tabOverride.getTabSize = function () {
449         return this.tabSize();
450     };
451 
452     /**
453      * Sets the tab size for all elements that have Tab Override enabled.
454      * 0 represents the tab character. The initial value is 0.
455      *
456      * @param  {Number}   [size=0]  the tab size
457      * @return {Function}           the tabOverride function
458      *
459      * @name setTabSize
460      * @function
461      * @memberOf jQuery.fn.tabOverride
462      * @deprecated since 2.0 - use tabSize() instead
463      */
464     tabOverride.setTabSize = function (size) {
465         return this.tabSize(size || 0);
466     };
467 
468     /**
469      * Gets or sets the auto indent setting. True if each line should be
470      * automatically indented (default = false).
471      *
472      * @param  {Boolean}          [enable]  whether auto indent should be enabled
473      * @return {Boolean|Function}           whether auto indent is enabled or the
474      *                                      tabOverride function
475      *
476      * @name autoIndent
477      * @function
478      * @memberOf jQuery.fn.tabOverride
479      */
480     tabOverride.autoIndent = TABOVERRIDE.autoIndent;
481 }(jQuery, TABOVERRIDE));
