 56     /**
 57      * Determines whether the specified modifier keys match the modifier keys
 58      * that were pressed.
 59      *
 60      * @param  {String[]} modifierKeys  the modifier keys to check - ex: ['shiftKey']
 61      * @param  {Event}    e             the event object for the keydown event
 62      * @return {Boolean}                whether modifierKeys are valid for the event
 63      *
 64      * @private
 65      */
 66     function isValidModifierKeyCombo(modifierKeys, e) {
 67         var modifierKeyNames = ['alt', 'ctrl', 'meta', 'shift'],
 68             numModKeys = modifierKeys.length,
 69             i,
 70             j,
 71             currModifierKey,
 72             isValid = true;
 73 
 74         // check that all required modifier keys were pressed
 75         for (i = 0; i < numModKeys; i += 1) {
 76             if (!e[modifierKeys[i]]) {
 77                 isValid = false;
 78                 break;
 79             }
 80         }
 81 
 82         // if requirements were met, check for additional modifier keys
 83         if (isValid) {
 84             for (i = 0; i < modifierKeyNames.length; i += 1) {
 85                 currModifierKey = modifierKeyNames[i] + 'Key';
 86 
 87                 // if this key was pressed
 88                 if (e[currModifierKey]) {
 89                     // if there are required keys, check whether the current key
 90                     // is required
 91                     if (numModKeys) {
 92                         isValid = false;
 93 
 94                         // if this is a required key, continue
 95                         for (j = 0; j < numModKeys; j += 1) {
 96                             if (currModifierKey === modifierKeys[j]) {
 97                                 isValid = true;
 98                                 break;
 99                             }
100                         }
101                     } else {
102                         // no required keys, but one was pressed
103                         isValid = false;
104                     }
105                 }
106 
107                 // an extra key was pressed, don't check anymore
108                 if (!isValid) {
109                     break;
110                 }
111             }
112         }
113 
114         return isValid;
115     }

143     /**
144      * Event handler to insert or remove tabs and newlines on the keyDown event
145      * for the tab or enter key.
146      *
147      * @param {Event} e  the event object
148      *
149      * @name overrideKeyDown
150      * @function
151      * @memberOf TABOVERRIDE
152      */
153     TABOVERRIDE.overrideKeyDown = function (e) {
154         e = e || event;
155 
156         // textarea elements can only contain text nodes which don't receive
157         // keydown events, so the event target/srcElement will always be the
158         // textarea element, however, prefer currentTarget in order to support
159         // delegated events in compliant browsers
160         var target = e.currentTarget || e.srcElement, // don't use the "this" keyword (doesn't work in old IE)
161             key = e.keyCode, // the key code for the key that was pressed
162             tab, // the string representing a tab
163             tabLen, // the length of a tab
164             text, // initial text in the textarea
165             range, // the IE TextRange object
166             tempRange, // used to calculate selection start and end positions in IE
167             preNewlines, // the number of newline character sequences before the selection start (for IE)
168             selNewlines, // the number of newline character sequences within the selection (for IE)
169             initScrollTop, // initial scrollTop value used to fix scrolling in Firefox
170             selStart, // the selection start position
171             selEnd, // the selection end position
172             sel, // the selected text
173             startLine, // for multi-line selections, the first character position of the first line
174             endLine, // for multi-line selections, the last character position of the last line
175             numTabs, // the number of tabs inserted / removed in the selection
176             startTab, // if a tab was removed from the start of the first line
177             preTab, // if a tab was removed before the start of the selection
178             whitespace, // the whitespace at the beginning of the first selected line
179             whitespaceLen, // the length of the whitespace at the beginning of the first selected line
180             CHARACTER = 'character'; // string constant used for the Range.move methods
181 
182         // don't do any unnecessary work
183         if ((target.nodeName && target.nodeName.toLowerCase() !== 'textarea') ||
184                 (key !== tabKey && key !== untabKey && (key !== 13 || !autoIndent))) {
185             return;
186         }
187 
188         // initialize variables used for tab and enter keys
189         inWhitespace = false; // this will be set to true if enter is pressed in the leading whitespace
190         text = target.value;
191 
192         // this is really just for Firefox, but will be used by all browsers that support
193         // selectionStart and selectionEnd - whenever the textarea value property is reset,
194         // Firefox scrolls back to the top - this is used to set it back to the original value
195         // scrollTop is nonstandard, but supported by all modern browsers
196         initScrollTop = target.scrollTop;
197 
198         // get the text selection
199         if (typeof target.selectionStart === 'number') {
200             selStart = target.selectionStart;
201             selEnd = target.selectionEnd;
202             sel = text.slice(selStart, selEnd);
203 
204         } else if (document.selection) { // IE
205             range = document.selection.createRange();
206             sel = range.text;
207             tempRange = range.duplicate();
208             tempRange.moveToElementText(target);
209             tempRange.setEndPoint('EndToEnd', range);
210             selEnd = tempRange.text.length;
211             selStart = selEnd - sel.length;
212 
213             // whenever the value of the textarea is changed, the range needs to be reset
214             // IE <9 (and Opera) use both \r and \n for newlines - this adds an extra character
215             // that needs to be accounted for when doing position calculations with ranges
216             // these values are used to offset the selection start and end positions
217             if (newlineLen > 1) {
218                 preNewlines = text.slice(0, selStart).split(newline).length - 1;
219                 selNewlines = sel.split(newline).length - 1;
220             } else {
221                 preNewlines = selNewlines = 0;
222             }
223         } else {
224             return; // cannot access textarea selection - do nothing
225         }
226 
227         // tab / untab key - insert / remove tab
228         if (key === tabKey || key === untabKey) {
229 
230             // initialize tab variables
231             tab = aTab;
232             tabLen = tab.length;
233             numTabs = 0;
234             startTab = 0;
235             preTab = 0;
236 
237             // multi-line selection
238             if (selStart !== selEnd && sel.indexOf('\n') !== -1) {
239                 // for multiple lines, only insert / remove tabs from the beginning of each line
240 
241                 // find the start of the first selected line
242                 if (selStart === 0 || text.charAt(selStart - 1) === '\n') {
243                     // the selection starts at the beginning of a line
244                     startLine = selStart;
245                 } else {
246                     // the selection starts after the beginning of a line
247                     // set startLine to the beginning of the first partially selected line
248                     // subtract 1 from selStart in case the cursor is at the newline character,
249                     // for instance, if the very end of the previous line was selected
250                     // add 1 to get the next character after the newline
251                     // if there is none before the selection, lastIndexOf returns -1
252                     // when 1 is added to that it becomes 0 and the first character is used
253                     startLine = text.lastIndexOf('\n', selStart - 1) + 1;
254                 }
255 
256                 // find the end of the last selected line
257                 if (selEnd === text.length || text.charAt(selEnd) === '\n') {
258                     // the selection ends at the end of a line
259                     endLine = selEnd;
260                 } else {
261                     // the selection ends before the end of a line
262                     // set endLine to the end of the last partially selected line
263                     endLine = text.indexOf('\n', selEnd);
264                     if (endLine === -1) {
265                         endLine = text.length;
266                     }
267                 }
268 
269                 // tab key combo - insert tabs
270                 if (tabKeyComboPressed(key, e)) {
271 
272                     numTabs = 1; // for the first tab
273 
274                     // insert tabs at the beginning of each line of the selection
275                     target.value = text.slice(0, startLine) + tab + text.slice(startLine, selStart) +
276                         sel.replace(/\n/g, function () {
277                             numTabs += 1;
278                             return '\n' + tab;
279                         }) + text.slice(selEnd);
280 
281                     // set start and end points
282                     if (range) { // IE
283                         range.collapse();
284                         range.moveEnd(CHARACTER, selEnd + (numTabs * tabLen) - selNewlines - preNewlines);
285                         range.moveStart(CHARACTER, selStart + tabLen - preNewlines);
286                         range.select();
287                     } else {
288                         // the selection start is always moved by 1 character
289                         target.selectionStart = selStart + tabLen;
290                         // move the selection end over by the total number of tabs inserted
291                         target.selectionEnd = selEnd + (numTabs * tabLen);
292                         target.scrollTop = initScrollTop;
293                     }
294                 } else if (untabKeyComboPressed(key, e)) {
295                     // if the untab key combo was pressed, remove tabs instead of inserting them
296 
297                     if (text.slice(startLine).indexOf(tab) === 0) {
298                         // is this tab part of the selection?
299                         if (startLine === selStart) {
300                             // it is, remove it
301                             sel = sel.slice(tabLen);
302                         } else {
303                             // the tab comes before the selection
304                             preTab = tabLen;
305                         }
306                         startTab = tabLen;
307                     }
308 
309                     target.value = text.slice(0, startLine) + text.slice(startLine + preTab, selStart) +
310                         sel.replace(new RegExp('\n' + tab, 'g'), function () {
311                             numTabs += 1;
312                             return '\n';
313                         }) + text.slice(selEnd);
314 
315                     // set start and end points
316                     if (range) { // IE
317                         // setting end first makes calculations easier
318                         range.collapse();
319                         range.moveEnd(CHARACTER, selEnd - startTab - (numTabs * tabLen) - selNewlines - preNewlines);
320                         range.moveStart(CHARACTER, selStart - preTab - preNewlines);
321                         range.select();
322                     } else {
323                         // set start first for Opera
324                         target.selectionStart = selStart - preTab; // preTab is 0 or tabLen
325                         // move the selection end over by the total number of tabs removed
326                         target.selectionEnd = selEnd - startTab - (numTabs * tabLen);
327                     }
328                 } else {
329                     return; // do nothing for invalid key combinations
330                 }
331 
332             } else { // single line selection
333 
334                 // tab key combo - insert a tab
335                 if (tabKeyComboPressed(key, e)) {
336                     if (range) { // IE
337                         range.text = tab;
338                         range.select();
339                     } else {
340                         target.value = text.slice(0, selStart) + tab + text.slice(selEnd);
341                         target.selectionEnd = target.selectionStart = selStart + tabLen;
342                         target.scrollTop = initScrollTop;
343                     }
344                 } else if (untabKeyComboPressed(key, e)) {
345                     // if the untab key combo was pressed, remove a tab instead of inserting one
346 
347                     // if the character before the selection is a tab, remove it
348                     if (text.slice(selStart - tabLen).indexOf(tab) === 0) {
349                         target.value = text.slice(0, selStart - tabLen) + text.slice(selStart);
350 
351                         // set start and end points
352                         if (range) { // IE
353                             // collapses range and moves it by -1 tab
354                             range.move(CHARACTER, selStart - tabLen - preNewlines);
355                             range.select();
356                         } else {
357                             target.selectionEnd = target.selectionStart = selStart - tabLen;
358                             target.scrollTop = initScrollTop;
359                         }
360                     }
361                 } else {
362                     return; // do nothing for invalid key combinations
363                 }
364             }
365         } else if (autoIndent) { // Enter key
366             // insert a newline and copy the whitespace from the beginning of the line
367 
368             // find the start of the first selected line
369             if (selStart === 0 || text.charAt(selStart - 1) === '\n') {
370                 // the selection starts at the beginning of a line
371                 // do nothing special
372                 inWhitespace = true;
373                 return;
374             }
375 
376             // see explanation under "multi-line selection" above
377             startLine = text.lastIndexOf('\n', selStart - 1) + 1;
378 
379             // find the end of the first selected line
380             endLine = text.indexOf('\n', selStart);
381 
382             // if no newline is found, set endLine to the end of the text
383             if (endLine === -1) {
384                 endLine = text.length;
385             }
386 
387             // get the whitespace at the beginning of the first selected line (spaces and tabs only)
388             whitespace = text.slice(startLine, endLine).match(/^[ \t]*/)[0];
389             whitespaceLen = whitespace.length;
390 
391             // the cursor (selStart) is in the whitespace at beginning of the line
392             // do nothing special
393             if (selStart < startLine + whitespaceLen) {
394                 inWhitespace = true;
395                 return;
396             }
397 
398             if (range) { // IE
399                 // insert the newline and whitespace
400                 range.text = '\n' + whitespace;
401                 range.select();
402             } else {
403                 // insert the newline and whitespace
404                 target.value = text.slice(0, selStart) + '\n' + whitespace + text.slice(selEnd);
405                 // Opera uses \r\n for a newline, instead of \n,
406                 // so use newlineLen instead of a hard-coded value
407                 target.selectionEnd = target.selectionStart = selStart + newlineLen + whitespaceLen;
408                 target.scrollTop = initScrollTop;
409             }
410         }
411 
412         if (e.preventDefault) {
413             e.preventDefault();
414         } else {
415             e.returnValue = false;
416             return false;
417         }
418     };
